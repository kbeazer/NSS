#!/bin/bash
# This is an example of using functions.
function greet {
	echo "Hi, $1 What a nice $2 today"
}

echo "And now, a greeting:"
greet Kareem day
greet Everyone evening

#This is a new function.
function numberthings {
	i=1
	for f in $@; do
		echo $i: $f
		((i+1))
	done
}

numberthings $(ls)

numberthings pine birch maple spruce
